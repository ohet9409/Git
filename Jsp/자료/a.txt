-- 테이블 생성
CREATE TABLE member(
    name VARCHAR2(30) not null,
    id VARCHAR2(30) PRIMARY key, -- 기본키, unique, not null - 무결성 제약 조건
    pwd VARCHAR2(30) not null,
    gender VARCHAR2(3),
    email1 VARCHAR2(20),
    email2 VARCHAR2(20),
    tel1 varchar2(10),
    tel2 varchar2(10),
    tel3 varchar2(10),
    addr VARCHAR2(100),
    logtime date
);
